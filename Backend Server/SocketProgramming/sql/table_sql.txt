CREATE TABLE ACN_LOCK_CONNECTION_INFO (
LOCKID VARCHAR(15) NOT NULL,
IS_BEING_USED CHAR DEFAULT 'N',
FOREIGN KEY (LOCKID) REFERENCES ACN_LOCK_INFO(LOCKID) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS ACN_LOCK_INFO (
  USERNAME varchar(15) NOT NULL,
  LOCKID varchar(15) NOT NULL,
  SOFT_LOCK_STATUS varchar(8) DEFAULT 'LOCKED',
  HARD_LOCK_STATUS varchar(8) DEFAULT 'LOCKED',
  STATUS_CHANGED timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  COMMENTS varchar(50) DEFAULT NULL,
  CONSTRAINT CONSTRAINT_UNIQUE_ACN_LOCK_INFO UNIQUE (USERNAME,LOCKID),
  PRIMARY KEY (LOCKID)
) DEFAULT CHARSET=latin1;



CREATE TABLE IF NOT EXISTS ACN_USER_INFO (
  USERNAME varchar(15) NOT NULL,
  PASSWORD varchar(15) NOT NULL,
  CREATED_AT timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (USERNAME)
) DEFAULT CHARSET=latin1;


CREATE TABLE  IF NOT EXISTS ACN_LOCK_LOCATION_INFO (
  USERNAME varchar(15) NOT NULL,
  LOCKID varchar(15) NOT NULL,
  HOME_LONGITUDE  
  HOME_LATITUDE 
  PRIMARY KEY (USERNAME)
) DEFAULT CHARSET=latin1;



SELECT A.COMMENTS AS COMMENTS, B.HOME_LONGITUDE AS HOME_LONGITUDE, B.HOME_LATITUDE AS HOME_LATITUDE 
FROM ACN_LOCK_INFO AS A, ACN_LOCK_LOCATION_INFO AS B WHERE A.LOCKID = B.LOCKID AND A.LOCKID = 'LOCK_2';
